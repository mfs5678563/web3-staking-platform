// SPDX-License-Identifier: MIT pragma solidity ^0.8.20; interface IERC20 { function transfer(address to, uint256 value) external returns (bool); function transferFrom(address from,address to,uint256 value) external returns (bool); function balanceOf(address account) external view returns (uint256); } contract DTMStaking { IERC20 public dtm; address public owner; uint256 public totalStaked; uint256 public rewardPerSecond; uint256 public accRewardPerShare; uint256 public lastUpdateTime; struct User { uint256 amount; uint256 rewardDebt; address inviter; } mapping(address => User) public users; modifier onlyOwner(){ require(msg.sender == owner,"not owner"); _; } constructor(address _dtm,uint256 _rps){ dtm = IERC20(_dtm); rewardPerSecond = _rps; lastUpdateTime = block.timestamp; owner = msg.sender; } // 更新池子 function updatePool() public { if(totalStaked == 0){ lastUpdateTime = block.timestamp; return; } uint256 time = block.timestamp - lastUpdateTime; uint256 reward = time * rewardPerSecond; accRewardPerShare += reward * 1e12 / totalStaked; lastUpdateTime = block.timestamp; } // 绑定邀请 function bind(address _inviter) external { require(_inviter != msg.sender, "self"); require(users[msg.sender].inviter == address(0), "bound"); users[msg.sender].inviter = _inviter; } // 质押 function stake(uint256 amount) external { User storage user = users[msg.sender]; updatePool(); // 发收益 if(user.amount > 0){ uint256 pending = user.amount * accRewardPerShare / 1e12 - user.rewardDebt; if(pending > 0){ _safeTransfer(msg.sender, pending); _inviteReward(msg.sender, pending); } } dtm.transferFrom(msg.sender, address(this), amount); user.amount += amount; totalStaked += amount; user.rewardDebt = user.amount * accRewardPerShare / 1e12; } // 解押 function unstake(uint256 amount) external { User storage user = users[msg.sender]; require(user.amount >= amount,"not enough"); updatePool(); uint256 pending = user.amount * accRewardPerShare / 1e12 - user.rewardDebt; if(pending > 0){ _safeTransfer(msg.sender, pending); _inviteReward(msg.sender, pending); } user.amount -= amount; totalStaked -= amount; dtm.transfer(msg.sender, amount); user.rewardDebt = user.amount * accRewardPerShare / 1e12; } // 领取 function claim() external { User storage user = users[msg.sender]; updatePool(); uint256 pending = user.amount * accRewardPerShare / 1e12 - user.rewardDebt; require(pending > 0,"no reward"); _safeTransfer(msg.sender, pending); _inviteReward(msg.sender, pending); user.rewardDebt = user.amount * accRewardPerShare / 1e12; } // 查询收益 function pendingReward(address _user) external view returns(uint256){ User storage user = users[_user]; uint256 tempAcc = accRewardPerShare; if(block.timestamp > lastUpdateTime && totalStaked != 0){ uint256 reward = (block.timestamp - lastUpdateTime) * rewardPerSecond; tempAcc += reward * 1e12 / totalStaked; } return user.amount * tempAcc / 1e12 - user.rewardDebt; } // 邀请奖励（5%） function _inviteReward(address userAddr,uint256 reward) internal { address inv = users[userAddr].inviter; if(inv != address(0)){ uint256 bonus = reward * 5 / 100; _safeTransfer(inv, bonus); } } function _safeTransfer(address to,uint256 amount) internal { uint256 bal = dtm.balanceOf(address(this)); if(amount > bal) amount = bal; dtm.transfer(to, amount); } // 注入奖励池 function deposit(uint256 amount) external onlyOwner { dtm.transferFrom(msg.sender,address(this),amount); } function setReward(uint256 r) external onlyOwner{ updatePool(); rewardPerSecond = r; } }
